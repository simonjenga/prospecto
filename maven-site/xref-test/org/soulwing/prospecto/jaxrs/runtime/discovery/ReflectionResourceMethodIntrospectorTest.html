306   private Expectations annotatedLocatorMethodExpectations(
307       final Method method, final boolean hasAbstractReturnType,
308       final boolean hasTemplateResolver)
309       throws Exception {
310     return new Expectations() {
311       {
312         allowing(reflectionService).getReturnType(method);
313         will(returnValue(MockResource.class));
314         oneOf(reflectionService).isAbstractType(MockResource.class);
315         will(returnValue(hasAbstractReturnType));
316         oneOf(reflectionService).getAnnotation(method, ReferencedBy.class);
317         will(returnValue(AnnotationUtils.referencedByAnnotation(Object.class)));
318         oneOf(reflectionService).getAnnotation(method, TemplateResolver.class);
319         will(returnValue(hasTemplateResolver ? methodTemplateResolver : null));
320       }
321     };
322   }
323 
324   private Expectations typeDescriptorExpectations(final Class<?> returnType,
325       final ModelPath modelPath, final boolean hasTemplateResolver)
326       throws Exception {
327     return new Expectations() {
328       {
329         oneOf(typeIntrospector).describe(
330             with(returnType),
331             with(PARENT_PATH + RESOURCE_PATH),
332             with(modelPath),
333             with(hasTemplateResolver ? methodTemplateResolver :
334                 typeTemplateResolver),
335             with(reflectionService));
336         will(returnValue(Collections.singletonList(typeDescriptor)));
337       }
338     };
339   }
340 
341   private Expectations annotatedSubresourceExpectations(
342       final Method method, final boolean hasTemplateResolver) throws Exception {
343     return new Expectations() {
344       {
345         oneOf(reflectionService).getReturnType(method);
346         will(returnValue(MockSubResource.class));
347         oneOf(reflectionService).isAbstractType(MockSubResource.class);
348         will(returnValue(false));
349         oneOf(reflectionService).getAnnotation(method, ReferencedBy.class);
350         will(returnValue(null));
351         oneOf(reflectionService).getAnnotation(method, TemplateResolver.class);
352         will(returnValue(hasTemplateResolver ? typeTemplateResolver : null));
353       }
354     };
355   }
356 
357   private Expectations resourceAnnotationExpectations(
358       final Method method, final boolean hasPath, final boolean hasHttpMethod)
359       throws Exception {
360     return new Expectations() {
361       {
362         oneOf(reflectionService).getAnnotation(method, Path.class);
363         will(returnValue(hasPath ?
364             AnnotationUtils.pathAnnotation(RESOURCE_PATH) : null));
365         oneOf(reflectionService).getAnnotation(method, GET.class);
366         will(returnValue(hasHttpMethod ?
367             AnnotationUtils.httpGETAnnotation() : null));
368         allowing(reflectionService).getAnnotation(method, POST.class);
369         will(returnValue(null));
370         allowing(reflectionService).getAnnotation(method, PUT.class);
371         will(returnValue(null));
372         allowing(reflectionService).getAnnotation(method, DELETE.class);
373         will(returnValue(null));
374         allowing(reflectionService).getAnnotation(method, HEAD.class);
375         will(returnValue(null));
376         allowing(reflectionService).getAnnotation(method, OPTIONS.class);
377         will(returnValue(null));
378       }
379     };
380   }

396   private Expectations methodDescriptorExpectations(
397       final Method method, final boolean hasTemplateResolver) throws Exception {
398     return new Expectations() {
399       {
400         oneOf(descriptorFactory).newDescriptor(
401             with(method),
402             with(PARENT_PATH + RESOURCE_PATH),
403             with(MODEL_PATH.concat(Object.class)),
404             with(any(PathTemplateResolver.class)));
405         will(returnValue(methodDescriptor));
406       }
407     };
408   }
