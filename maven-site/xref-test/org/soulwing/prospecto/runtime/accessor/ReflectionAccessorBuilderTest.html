19  package org.soulwing.prospecto.runtime.accessor;
20  
21  import static org.hamcrest.MatcherAssert.assertThat;
22  import static org.hamcrest.Matchers.is;
23  import static org.hamcrest.Matchers.sameInstance;
24  
25  import org.junit.Test;
26  import org.soulwing.prospecto.api.AccessType;
27  import org.soulwing.prospecto.api.ViewTemplateException;
28  
29  /**
30   * Unit tests for {@link ReflectionAccessorBuilder}.
31   *
32   * @author Carl Harris
33   */
34  public class ReflectionAccessorBuilderTest {
35  
36    private static final Object PRIVATE_FIELD_VALUE = new Object();
37    private static final Object PUBLIC_FIELD_VALUE = new Object();
38    private static final Object PUBLIC_METHOD_VALUE = new Object();
39    private static final Object PRIVATE_METHOD_VALUE = new Object();
40  
41    private ReflectionAccessorBuilder accessorBuilder =
42        new ReflectionAccessorBuilder(MockModel.class);
43  
44    private MockModel model = new MockModel();
45  
46    @Test
47    public void testPrivateFieldAccessor() throws Exception {
48      final Accessor accessor = accessorBuilder
49          .propertyName("privateField")
50          .accessType(AccessType.FIELD)
51          .build();
52      assertThat(accessor.get(model), is(sameInstance(PRIVATE_FIELD_VALUE)));
53    }
54  
55    @Test
56    public void testPublicFieldAccessor() throws Exception {
57      final Accessor accessor = accessorBuilder
58          .propertyName("publicField")
59          .accessType(AccessType.FIELD)
60          .build();
61      assertThat(accessor.get(model), is(sameInstance(PUBLIC_FIELD_VALUE)));
62    }
63  
64    @Test
65    public void testProtectedFieldInSuperAccessor() throws Exception {
66      final Accessor accessor = accessorBuilder
67          .propertyName("privateFieldInSuper")
68          .accessType(AccessType.FIELD)
69          .build();
70      assertThat(accessor.get(model), is(sameInstance(PRIVATE_FIELD_VALUE)));
71    }
72  
73    @Test
74    public void testPublicFieldInSuperAccessor() throws Exception {
75      final Accessor accessor = accessorBuilder
76          .propertyName("publicFieldInSuper")
77          .accessType(AccessType.FIELD)
78          .build();
79      assertThat(accessor.get(model), is(sameInstance(PUBLIC_FIELD_VALUE)));
80    }
81  
82    @Test
83    public void testPublicMethodAccessor() throws Exception {
84      final Accessor accessor = accessorBuilder
85          .propertyName("publicMethod")
86          .accessType(AccessType.PROPERTY)
87          .build();
88  
89      assertThat(accessor.get(model), is(sameInstance(PUBLIC_METHOD_VALUE)));
90    }
91  
92    @Test(expected = ViewTemplateException.class)
93    public void testPrivateMethodAccessor() throws Exception {
94      // this isn't supported but could be; need to do our own accessor method
95      // introspection. Does anyone care?
96      accessorBuilder
97          .propertyName("privateMethod")
98          .accessType(AccessType.PROPERTY)
99          .build();
100   }
101 
102   @SuppressWarnings("unused")
103   abstract class MockSuperModel {
104 
105     private Object privateFieldInSuper = PRIVATE_FIELD_VALUE;
106 
107     public Object publicFieldInSuper = PUBLIC_FIELD_VALUE;
108 
109   }
110 
111   @SuppressWarnings("unused")
112   public class MockModel extends MockSuperModel {
113 
114     private Object privateField = PRIVATE_FIELD_VALUE;
115     public Object publicField = PUBLIC_FIELD_VALUE;
116 
117     Object getPrivateMethod() {
118       return PRIVATE_METHOD_VALUE;
119     }
120 
121     public Object getPublicMethod() {
122       return PUBLIC_METHOD_VALUE;
123     }
124 
125     public void setPublicMethod(Object value) { }
126 
127   }
128 
129 }
