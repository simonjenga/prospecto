127   @Test
128   public void testScopesPutAndGetByType() throws Exception {
129     final MockScope0Type scope0Mock = new MockScope0Type() {};
130 
131     final MutableScope scope0 = viewContext.newScope();
132     viewContext.getScopes().append(scope0);
133     scope0.put(SCOPE0);
134     scope0.put(scope0Mock);
135     assertThat(viewContext.get(String.class), is(sameInstance(SCOPE0)));
136     assertThat(viewContext.get(MockScope0Type.class), is(sameInstance(scope0Mock)));
137 
138     final MockScope1Type scope1Mock = new MockScope1Type() {};
139 
140     viewContext.push(SCOPE1, null);
141     viewContext.put(SCOPE1);
142     viewContext.put(scope1Mock);
143     assertThat(viewContext.get(String.class), is(sameInstance(SCOPE1)));
144     assertThat(viewContext.get(MockScope1Type.class), is(sameInstance(scope1Mock)));
145     assertThat(viewContext.get(MockScope0Type.class), is(sameInstance(scope0Mock)));
146 
147     final MockScope2Type scope2Mock = new MockScope2Type() {};
148 
149     viewContext.push(SCOPE2, null);
150     viewContext.put(SCOPE2);
151     viewContext.put(scope2Mock);
152     assertThat(viewContext.get(String.class), is(sameInstance(SCOPE2)));
153     assertThat(viewContext.get(MockScope2Type.class), is(sameInstance(scope2Mock)));
154     assertThat(viewContext.get(MockScope1Type.class), is(sameInstance(scope1Mock)));
155     assertThat(viewContext.get(MockScope0Type.class), is(sameInstance(scope0Mock)));
156 
157     viewContext.pop();
158     assertThat(viewContext.get(String.class), is(sameInstance(SCOPE1)));
159     assertThat(viewContext.get(MockScope1Type.class), is(sameInstance(scope1Mock)));
160     assertThat(viewContext.get(MockScope0Type.class), is(sameInstance(scope0Mock)));
161 
162     viewContext.pop();
163     assertThat(viewContext.get(String.class), is(sameInstance(SCOPE0)));
164     assertThat(viewContext.get(MockScope0Type.class), is(sameInstance(scope0Mock)));
165   }

343   @Test
344   public void testCopy() throws Exception {
345     final Scope scope =
346         context.mock(Scope.class);
347     final ViewListener listener =
348         context.mock(ViewListener.class);
349     final ValueTypeConverter valueTypeConverter =
350         context.mock(ValueTypeConverter.class);
351     final ReferenceResolver resolver =
352         context.mock(ReferenceResolver.class);
353     final AssociationManager associationManager =
354         context.mock(AssociationManager.class);
355 
356 
357     viewContext.getScopes().append(scope);
358     viewContext.getListeners().append(listener);
359     viewContext.getValueTypeConverters().append(valueTypeConverter);
360     viewContext.getReferenceResolvers().append(resolver);
361     viewContext.getAssociationManagers().append(associationManager);
362 
363     ViewContext contextCopy = new ConcreteViewContext(viewContext);
364     viewContext.getScopes().toList().clear();
365     viewContext.getListeners().toList().clear();
366     viewContext.getValueTypeConverters().toList().clear();
367     viewContext.getReferenceResolvers().toList().clear();
368     viewContext.getAssociationManagers().toList().clear();
369 
370     assertThat(contextCopy.getScopes().toList(), contains(scope));
371     assertThat(contextCopy.getListeners().toList(), contains(listener));
372     assertThat(contextCopy.getValueTypeConverters().toList(),
373         contains(valueTypeConverter));
374     assertThat(contextCopy.getReferenceResolvers().toList(),
375         contains(resolver));
376     assertThat(contextCopy.getAssociationManagers().toList(),
377         contains(associationManager));
378     assertThat(contextCopy.getOptions(), is(sameInstance(options)));
379   }
