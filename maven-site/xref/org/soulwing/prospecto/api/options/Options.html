19  package org.soulwing.prospecto.api.options;
20  
21  import java.util.Map;
22  
23  /**
24   * A map-like API for accessing context options.
25   *
26   * @author Carl Harris
27   */
28  public interface Options {
29  
30    /**
31     * Gets the value of the named option.
32     * @param name option name
33     * @return value or {@code null} if the option is not set
34     */
35    Object get(String name);
36  
37    /**
38     * Gets the value of the named option.
39     * @param name option name
40     * @param defaultValue default value
41     * @return option value or {@code defaultValue} if the option is not set
42     */
43    Object get(String name, Object defaultValue);
44  
45    /**
46     * Puts (sets) the value of the named option.
47     * @param name option name
48     * @param value the value to put
49     */
50    void put(String name, Object value);
51  
52    /**
53     * Tests whether a two-state (boolean) option is set to {@code true}.
54     * @param name option name
55     * @return {@code true} if the named option is set to {@code true}
56     * @throws ClassCastException if the option cannot be coerced to a boolean
57     */
58    boolean isEnabled(String name);
59  
60    /**
61     * Coerces this options collection to a map.
62     * <p>
63     * The returned map can be manipulated to modify this options collection.
64     * @return map
65     */
66    Map<String, Object> toMap();
67  
68  }
