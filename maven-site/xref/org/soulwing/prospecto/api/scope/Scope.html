19  package org.soulwing.prospecto.api.scope;
20  
21  /**
22   * An object that acts as a container for other objects and supports retrieval
23   * by type and by name.
24   *
25   * @author Carl Harris
26   */
27  public interface Scope {
28  
29    /**
30     * Gets a singleton object of the given type.
31     * @param type the type of object to retrieve
32     * @return singleton instance of the given type or {@code null} if there
33     *    is no object of the given type
34     * @throws IllegalStateException if there is more than one object of the
35     *    given type
36     */
37    <T> T get(Class<T> type);
38  
39    /**
40     * Gets a named object of the given type.
41     * @param name name of the object to retrieve
42     * @param type expected type of the named object
43     * @return instance of the given type that corresponds to the given name
44     *    or {@code null} if no such instance exists
45     * @throws ClassCastException if the object with the given name does not
46     *    have the expected type
47     */
48    <T> T get(String name, Class<T> type);
49  
50  }
