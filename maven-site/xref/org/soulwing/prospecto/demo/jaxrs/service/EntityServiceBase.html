19  package org.soulwing.prospecto.demo.jaxrs.service;
20  
21  import javax.persistence.EntityManager;
22  import javax.persistence.OptimisticLockException;
23  
24  import org.soulwing.prospecto.api.View;
25  import org.soulwing.prospecto.api.ViewApplicator;
26  import org.soulwing.prospecto.api.ViewContext;
27  import org.soulwing.prospecto.api.ViewTemplate;
28  import org.soulwing.prospecto.demo.jaxrs.domain.AbstractEntity;
29  import org.soulwing.prospecto.demo.jaxrs.domain.Contact;
30  
31  /**
32   * A base for entity service implementations.
33   * @author Carl Harris
34   */
35  public class EntityServiceBase<T extends AbstractEntity>
36      implements EntityService<T> {
37  
38    private final Class<T> entityClass;
39    private final ViewTemplate detailView;
40  
41    private ViewContext viewContext;
42    private EntityManager entityManager;
43  
44    EntityServiceBase(Class<T> entityClass, ViewTemplate detailView) {
45      this.entityClass = entityClass;
46      this.detailView = detailView;
47    }
48  
49    @Override
50    public View findById(Long id) throws NoSuchEntityException {
51      Object entity = entityManager.find(entityClass, id);
52      if (entity == null) {
53        throw new NoSuchEntityException(entityClass, id);
54      }
55      return detailView.generateView(entity, viewContext);
56    }
57  
58    @Override
59    @SuppressWarnings("unchecked")
60    public T newEntity(View view) {
61      return (T) detailView.createApplicator(view, viewContext).create();
62    }
63  
64    @Override
65    public Object create(View view) {
66      final T entity = newEntity(view);
67      entityManager.persist(entity);
68      entityManager.flush();
69      return entity.getId();
70    }
71  
72    @Override
73    public View update(Long id, View view) throws UpdateConflictException,
74        NoSuchEntityException {
75  
76      Object entity = entityManager.find(entityClass, id);
77      if (entity == null) {
78        throw new NoSuchEntityException(entityClass, id);
79      }
80  
81      try {
82        ViewApplicator applicator = detailView.createApplicator(view, viewContext);
83        applicator.update(entity);
84        entityManager.detach(entity);
85        entity = entityManager.merge(entity);
86        entityManager.flush();
87        return detailView.generateView(entity, viewContext);
88      }
89      catch (OptimisticLockException ex) {
90        throw new UpdateConflictException(Contact.class, id, ex);
91      }
92    }
93  
94    @Override
95    public void delete(Long id) {
96      T entity = entityManager.find(entityClass, id);
97      if (entity == null) return;
98      onDelete(entity);
99      entityManager.remove(entity);
100     entityManager.flush();
101   }
102 
103   protected void onDelete(T entity) {
104   }
105 
106   public ViewContext getViewContext() {
107     return viewContext;
108   }
109 
110   public void setViewContext(ViewContext viewContext) {
111     this.viewContext = viewContext;
112   }
113 
114   public EntityManager getEntityManager() {
115     return entityManager;
116   }
117 
118   public void setEntityManager(EntityManager entityManager) {
119     this.entityManager = entityManager;
120   }
121 
122 }
