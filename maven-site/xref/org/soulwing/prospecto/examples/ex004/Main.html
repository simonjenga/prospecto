19  package org.soulwing.prospecto.examples.ex004;
20  
21  import java.io.InputStream;
22  import java.io.OutputStream;
23  import java.util.Iterator;
24  import java.util.UUID;
25  
26  import org.soulwing.prospecto.ViewContextProducer;
27  import org.soulwing.prospecto.ViewReaderFactoryProducer;
28  import org.soulwing.prospecto.ViewTemplateBuilderProducer;
29  import org.soulwing.prospecto.ViewWriterFactoryProducer;
30  import org.soulwing.prospecto.api.View;
31  import org.soulwing.prospecto.api.ViewApplicator;
32  import org.soulwing.prospecto.api.ViewContext;
33  import org.soulwing.prospecto.api.ViewReader;
34  import org.soulwing.prospecto.api.ViewReaderFactory;
35  import org.soulwing.prospecto.api.ViewTemplate;
36  import org.soulwing.prospecto.api.ViewWriter;
37  import org.soulwing.prospecto.api.ViewWriterFactory;
38  import org.soulwing.prospecto.examples.util.ResourceUtil;
39  
40  class Main implements Runnable {
41  
42    static final ViewTemplate CONFERENCE_VIEW = ViewTemplateBuilderProducer
43        .object(Conference.class)
44            .value("id")
45            .value("name")
46            .value("abbreviation")
47            .end()
48        .build();
49  
50    static final ViewTemplate LEAGUE_VIEW = ViewTemplateBuilderProducer
51        .object(League.class)
52            .value("name")
53            .arrayOfObjects("conferences", CONFERENCE_VIEW)
54            .end()
55        .build();
56  
57    private final String format;
58    private final ViewReaderFactory readerFactory;
59    private final ViewWriterFactory writerFactory;
60    private final ViewContext context;
61  
62    Main(String format) {
63      this.format = format;
64      this.readerFactory = ViewReaderFactoryProducer.getFactory(format);
65      this.writerFactory = ViewWriterFactoryProducer.getFactory(format);
66      this.context = ViewContextProducer.newContext();
67    }
68  
69    public static void main(String[] args) {
70      new Main("json").run();
71      new Main("xml").run();
72    }
73  
74    public void run() {
75      System.out.println("*** Using " + format.toUpperCase() + " Format ****");
76  
77      League league = new League();
78      league.setName("National Collegiate Athletic Association Softball");
79  
80      Conference mockConference = new Conference();
81      mockConference.setId(UUID.fromString("827f8a6c-03c1-11e6-963b-03a0eba8ec15"));
82      mockConference.setName("Mock Conference");
83      league.getConferences().add(mockConference);
84  
85      Conference secConference = new Conference();
86      secConference.setId(UUID.fromString("6081d6e2-03bf-11e6-b087-43bbf6e56012"));
87      secConference.setName("Southeastern Conference");
88      league.getConferences().add(secConference);
89  
90      Conference accConference = new Conference();
91      accConference.setId(UUID.fromString("74ca21a4-03bf-11e6-b9d2-038e70fcd31b"));
92      accConference.setName("Atlantic Coast Conference");
93      league.getConferences().add(accConference);
94  
95      generateView(league, System.out);
96  
97      final InputStream inputStream = ResourceUtil.openStream(
98          Main.class, "revisedView." + format);
99  
100     applyView(inputStream, league);
101 
102     assert league.getName().equals("NCAA Softball");
103 
104     Iterator<Conference> conferences = league.getConferences().iterator();
105     assert conferences.hasNext();
106     assert conferences.next().getAbbreviation().equals("SEC");
107     assert conferences.hasNext();
108     assert conferences.next().getAbbreviation().equals("ACC");
109     assert conferences.hasNext();
110     assert conferences.next().getAbbreviation().equals("Big10");
111 
112     generateView(league, System.out);
113 
114   }
115 
116   private void generateView(League league, OutputStream outputStream) {
117 
118     View view = LEAGUE_VIEW.generateView(league, context);
119 
120     ViewWriter writer = writerFactory.newWriter(view, outputStream);
121 
122     writer.writeView();
123     System.out.println();
124   }
125 
126   private void applyView(InputStream inputStream, League league) {
127 
128     ViewReader reader = readerFactory.newReader(inputStream);
129     View revisedView = reader.readView();
130 
131     ViewApplicator applicator = LEAGUE_VIEW.createApplicator(revisedView,
132         context);
133 
134     applicator.update(league);
135   }
136 
137 }
