19  package org.soulwing.prospecto.jaxrs.runtime.discovery;
20  
21  import java.util.ArrayList;
22  import java.util.Collection;
23  import java.util.Set;
24  
25  import javax.ws.rs.Path;
26  import javax.ws.rs.core.UriBuilder;
27  
28  import org.slf4j.Logger;
29  import org.slf4j.LoggerFactory;
30  import org.soulwing.prospecto.jaxrs.runtime.path.ModelPath;
31  import org.soulwing.prospecto.jaxrs.runtime.ReflectionService;
32  import org.soulwing.prospecto.jaxrs.runtime.ResourceConfigurationException;
33  import org.soulwing.prospecto.jaxrs.runtime.ResourceDescriptor;
34  import org.soulwing.prospecto.jaxrs.runtime.ResourceDiscoveryService;
35  import org.soulwing.prospecto.jaxrs.api.TemplateResolver;
36  
37  /**
38   * A concrete {@link ResourceDiscoveryService}.
39   *
40   * @author Carl Harris
41   */
42  public class ReflectionResourceDiscoveryService
43      implements ResourceDiscoveryService {
44  
45    private static final Logger logger = LoggerFactory.getLogger(
46        ReflectionResourceDiscoveryService.class);
47  
48    private final ResourceTypeIntrospector typeIntrospector;
49  
50    /**
51     * Constructs a new instance that will use the default type introspector.
52     */
53    public ReflectionResourceDiscoveryService() {
54      this(new ReflectionResourceTypeIntrospector());
55    }
56  
57    /**
58     * Constructs a new instance that will use the given type introspector.
59     * @param typeIntrospector type introspector
60     */
61    ReflectionResourceDiscoveryService(ResourceTypeIntrospector typeIntrospector) {
62      this.typeIntrospector = typeIntrospector;
63    }
64  
65    @Override
66    public Collection<ResourceDescriptor> discoverResources(
67        String applicationPath, ReflectionService reflectionService)
68        throws ResourceConfigurationException {
69  
70      logger.debug("resource discovery started");
71      final ModelPath modelPath = ModelPath.with();
72      final Set<Class<?>> rootResourceTypes =
73          reflectionService.getTypesAnnotatedWith(Path.class);
74      final Collection<ResourceDescriptor> descriptors = new ArrayList<>();
75  
76      for (Class<?> rootResourceType : rootResourceTypes) {
77        final Path path = reflectionService.getAnnotation(rootResourceType,
78            Path.class);
79  
80        final String qualifiedPath = UriBuilder.fromPath(applicationPath)
81            .path(path.value()).toTemplate();
82  
83        logger.trace("discovered root resource {}",
84            rootResourceType.getSimpleName());
85  
86        TemplateResolver templateResolver = reflectionService.getAnnotation(
87            rootResourceType, TemplateResolver.class);
88  
89        descriptors.addAll(typeIntrospector.describe(rootResourceType,
90            qualifiedPath, modelPath,
91            templateResolver, reflectionService));
92      }
93      logger.debug("resource discovery completed");
94      return descriptors;
95    }
96  
97  }
