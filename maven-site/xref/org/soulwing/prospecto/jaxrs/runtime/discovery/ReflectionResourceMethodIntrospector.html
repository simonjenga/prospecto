47  /**
48   * A {@link ResourceMethodIntrospector} that utilizes reflection.
49   *
50   * @author Carl Harris
51   */
52  class ReflectionResourceMethodIntrospector
53      implements ResourceMethodIntrospector {
54  
55    private static final Logger logger = LoggerFactory.getLogger(
56        ReflectionResourceMethodIntrospector.class);
57  
58  
59    private final ResourceDescriptorFactory descriptorFactory;
60  
61    /**
62     * Constructs a new instance using the default resource descriptor factory.
63     */
64    ReflectionResourceMethodIntrospector() {
65      this(new SimpleResourceDescriptorFactory());
66    }
67  
68    /**
69     * Constructs a new instance using the specified resource descriptor factory.
70     * @param descriptorFactory descriptor factory
71     */
72    ReflectionResourceMethodIntrospector(
73        ResourceDescriptorFactory descriptorFactory) {
74      this.descriptorFactory = descriptorFactory;
75    }
76  
77    @Override
78    public Collection<ResourceDescriptor> describe(Method method,
79        String resourcePath, ModelPath modelPath,
80        TemplateResolver templateResolver,
81        ReflectionService reflectionService,
82        ResourceTypeIntrospector typeIntrospector)
83        throws ResourceConfigurationException {
84  
85      final Path path = reflectionService.getAnnotation(method, Path.class);
86      resourcePath = resourcePath(resourcePath, path);
87  
88      final boolean resourceMethod = isResourceMethod(method, reflectionService);
89  
90      if (path == null && !resourceMethod) {
91        return Collections.emptyList();
92      }
93  
94      ReferencedBy referencedBy = reflectionService.getAnnotation(method,
95          ReferencedBy.class);
96  
97      TemplateResolver methodTemplateResolver = reflectionService.getAnnotation(
98          method, TemplateResolver.class);
99  
100     Class<?> returnType = reflectionService.getReturnType(method);
101 
102     if (resourceMethod) {
103       if (referencedBy == null) {
104         logger.trace("ignoring method {}", methodToString(method));
105         return Collections.emptyList();
106       }
107 
108       if (referencedBy.inherit()) {
109         modelPath = modelPath.concat(referencedBy);
110       }
111 
112       if (methodTemplateResolver != null) {
113         templateResolver = methodTemplateResolver;
114       }
115 
116       if (templateResolver == null) {
117         throw new ResourceConfigurationException(
118             "no template resolver for method " + methodToString(method));
119       }
120 
121       final PathTemplateResolver pathTemplateResolver = TemplateResolverUtils
122           .newResolver(templateResolver.value());
123 
124       return Collections.singletonList(descriptorFactory.newDescriptor(method,
125           resourcePath, modelPath, pathTemplateResolver));
126     }
127 
128 
129     if (reflectionService.isAbstractType(returnType)) {
130       if (referencedBy == null) return Collections.emptyList();
131 
132       returnType = findMatchingSubResourceType(modelPath.concat(referencedBy),
133           reflectionService, method);
134     }
135     else if (referencedBy != null && referencedBy.inherit()) {
136       modelPath = modelPath.concat(referencedBy);
137     }
138 
139     if (methodTemplateResolver == null) {
140       TemplateResolver typeTemplateResolver = reflectionService.getAnnotation(
141           returnType, TemplateResolver.class);
142       if (typeTemplateResolver != null) {
143         templateResolver = typeTemplateResolver;
144       }
145     }
146     else {
147       templateResolver = methodTemplateResolver;
148     }
149 
150     return typeIntrospector.describe(returnType, resourcePath, modelPath,
151         templateResolver, reflectionService);
152   }
153 
154   /**
155    * Determines whether the given method is annotated with an HTTP method
156    * annotation.
157    * @param method the method to examine
158    * @param reflector reflection service
159    * @return {@code true} if {@code method} has at least one HTTP method
160    *    annotation
161    */
162   private boolean isResourceMethod(Method method,
163       ReflectionService reflector) {
164     if (reflector.getAnnotation(method, GET.class) != null) return true;
165     if (reflector.getAnnotation(method, POST.class) != null) return true;
166     if (reflector.getAnnotation(method, PUT.class) != null) return true;
167     if (reflector.getAnnotation(method, DELETE.class) != null) return true;
168     if (reflector.getAnnotation(method, HEAD.class) != null) return true;
169     if (reflector.getAnnotation(method, OPTIONS.class) != null) return true;
170     return false;
171   }
172 
173   /**
174    * Finds the unique concrete subtype of the return type of the given method
175    * whose referenced-by annotation is equivalent to the referenced-by
176    * annotation of the method.
177    *
178    * @param modelPath model path to match
179    * @param reflectionService reflection service  @return matching sub type
180    * @param method the subject method
181    * @throws ResourceConfigurationException if the number of concrete types
182    *    that satisfy the above criterion is not equal to 1
183    */
184   private Class<?> findMatchingSubResourceType(ModelPath modelPath,
185       ReflectionService reflectionService, Method method) {
186 
187     final List<Class<?>> types = new ArrayList<>();
188     final Class<?> returnType = reflectionService.getReturnType(method);
189     final Class<?>[] methodReferences = modelPath.asArray();
190     for (Class<?> type : reflectionService.getSubTypesOf(returnType)) {
191       ReferencedBy typeReferencedBy = reflectionService.getAnnotation(type,
192           ReferencedBy.class);
193       if (typeReferencedBy == null) continue;
194       final Class<?>[] typeReferences = typeReferencedBy.value();
195       if (!Arrays.equals(typeReferences, methodReferences)) continue;
196       types.add(type);
197     }
198 
199     final int numTypes = types.size();
200     if (numTypes == 0) {
201       throw new ResourceConfigurationException("there is no subtype of "
202           + returnType.getSimpleName() + " with a @"
203           + ReferencedBy.class.getSimpleName() + " that matches "
204           + modelPath + " at " + methodToString(method));
205     }
206     else if (numTypes > 1) {
207       throw new ResourceConfigurationException(
208           "there is more than one subtype of "
209           + returnType.getSimpleName() + " with a @"
210           + ReferencedBy.class.getSimpleName() + " that matches "
211           + modelPath + " at " + method);
212     }
213 
214     return types.get(0);
215 
216   }
217 
218   /**
219    * Creates a new resource path from a parent path and the path specified
220    * by a {@link Path} annotation.
221    * @param parent parent path
222    * @param path path annotation whose value is to be appended
223    * @return {@code parent} with the value of {@link Path} appended to it
224    */
225   private String resourcePath(String parent, Path path) {
226     UriBuilder uriBuilder = UriBuilder.fromUri(parent);
227     if (path != null) {
228       uriBuilder.path(path.value());
229     }
230     return uriBuilder.toTemplate();
231   }
232 
233 
234   private String methodToString(Method method) {
235     return method.getDeclaringClass().getSimpleName() + "." + method.getName();
236   }
